### 魔法攻撃の実装

```typescript
function useMagicAttack(skillType: SkillType, position: Vector2) {
  let damage = 0
  let range = 0
  
  switch (skillType) {
    case SkillType.FIREBALL:
      damage = 100
      range = 80
      showExplosion(position, range)
      break
      
    case SkillType.THUNDERBOLT:
      damage = 200
      range = 120
      showExplosion(position, range)
      showSkillAura(position, '#fbbf24', 0.5)
      break
  }
  
  // 範囲内の敵にダメージ
  monsters.forEach(monster => {
    if (distance(monster.position, position) < range) {
      monster.takeDamage(damage)
      showDamageText(monster.position, damage, true)
    }
  })
}
```

getAttackPower(): number {
  let attack = this.stats.attack
  
  // ファーストストライク効果
  if (this.firstStrikeActive) {
    attack *= 3  // クリティカル
