### 魔法攻撃システム
```typescript
private handleMagicAttack(skillType: string, position: Vector2): void {
  let damage = 0
  let range = 0

  switch (skillType) {
    case 'FIREBALL':
      damage = 100
      range = 80
      break
    case 'THUNDERBOLT':
      damage = 200
      range = 120
      break
  }

  // 範囲内のモンスターにダメージ
  this.monsters.forEach(monster => {
    const distance = monster.distanceTo(position)
    if (distance < range) {
      monster.takeDamage(damage)
      // エフェクト表示
    }
  })
}
```

getAttackPower(): number {
